class Solution{
  public:
  int helper(const vector<int>& a, int n, long long k){
      long long int pro=1;
      int tot =0;
    int l =0,r=0;
    
    while(r < n){
        
        pro *= a[r];
        if(pro >= k){
            while(pro >= k && l<=r){
                pro = pro/a[l];
                l++;
            }
            
        }
         tot += ((r-l)+1);
            r++;
    }
    return tot;
   
  }
    int countSubArrayProductLessThanK(const vector<int>& a, int n, long long k) {
        return helper(a,n,k);   
    }
};

int main() {
    int t;

        int n, i;
        long long int k;
        cin >> n >> k;
        vector<int> arr(n);
        for (i = 0; i < n; i++) cin >> arr[i];
        Solution obj;
        cout << obj.countSubArrayProductLessThanK(arr, n, k) << endl;
    }
    return 0;
}
